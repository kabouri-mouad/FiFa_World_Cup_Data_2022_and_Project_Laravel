	J2.NOM_JOUEUR as NOM_JOUEUR2,
	M.NOM_PHASE,
	E.NOM_EQUIPE,
	MINUTE_CHANGEMENT
FROM CHANGEMENT C
INNER JOIN JOUEURS J1 ON J1.ID_JOUEUR = C.ID_JOUEUR1
INNER JOIN JOUEURS J2 ON J2.ID_JOUEUR = C.ID_JOUEUR2
INNER JOIN MATCHS M ON M.ID_MATCH = C.ID_MATCH
INNER JOIN EQUIPES E ON E.ID_EQUIPE = C.ID_EQUIPE;
